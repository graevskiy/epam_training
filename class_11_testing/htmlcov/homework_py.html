44 :return: ratio of base rates

60 def to(self, other): 

61 """Creates instance of `other` currency from a given `self` 

62 re-calculates amount from `self` currency to `other` 

63 

64 :param other: a class which instance we'll create and return 

65 :type other: subclass of `Currency` 

66 :return: object of `other` from `self` 

67 :rtype: instance of `other's` class 

68 """ 

69 return other(self.amount * type(self).exchange_rate(other))

71 def convert(self, other): 

72 """Does inversion of `to` method. Converts instance `other` 

73 to instance of `self`. More of less is used internally for 

74 comparison and addition magic methods. 

75 

76 :param other: another instance of currency we want to convert 

77 :type other: instance of subclass of `Currency` 

78 :return: converted currency item 

79 :rtype: instance of class of `self`

89 def __add__(self, other): 

90 """Re-load magic method to be able to add items 

91 Check if `other` is an instance of a subclass. 

92 If yes, add their `amounts` and return new instance. 

93 If not, assume `other` is addable (like int or float) 

94 and do straight addition 

95 

96 :param other: something we add to `self`

98 :return: new item with updated (added) amount

101 if isinstance(other, Currency):
